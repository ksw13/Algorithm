class Solution {
    int[][] answer;
    int index=0;
    
    public void hanoi(int n, int start, int end){
        if(n==1){
            answer[index][0]=start;
            answer[index][1]=end;
            index++;
            return;
        }
        
        hanoi(n-1,start,6-start-end);
        answer[index][0]=start;
        answer[index][1]=end;
        index++;
        hanoi(n-1,6-start-end,end);
    }
    
    public int[][] solution(int n) {
        answer = new int[(int)Math.pow(2,n)-1][2];
        
        hanoi(n,1,3);
        
        return answer;
    }
}
